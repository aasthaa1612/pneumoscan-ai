// ─── Gauge SVG ────────────────────────────────────────────────────────────────
function Gauge({ value, color, label, size = 110 }) {
    const r = 38;
    const cx = 55;
    const cy = 55;
    const circumference = Math.PI * r;
    const offset = circumference - (value / 100) * circumference;
    return (
        <div style={{ display: "flex", flexDirection: "column", alignItems: "center", gap: 4 }}>
            <svg width={size} height={size * 0.6} viewBox="0 0 110 62">
                <path d={`M 17 55 A 38 38 0 0 1 93 55`} fill="none" stroke="rgba(110,165,255,0.10)" strokeWidth="8" strokeLinecap="round" />
                <path
                    d={`M 17 55 A 38 38 0 0 1 93 55`}
                    fill="none"
                    stroke={color}
                    strokeWidth="8"
                    strokeLinecap="round"
                    strokeDasharray={circumference}
                    strokeDashoffset={offset}
                    style={{ transition: "stroke-dashoffset 1.2s cubic-bezier(0.34,1.56,0.64,1)", filter: `drop-shadow(0 0 6px ${color}88)` }}
                />
                <text x="55" y="52" textAnchor="middle" fill={color} fontSize="15" fontWeight="700" fontFamily="'JetBrains Mono',monospace">
                    {value}%
                </text>
            </svg>
            <span style={{ fontFamily: "'JetBrains Mono',monospace", fontSize: "0.6rem", color: "var(--text3)", textTransform: "uppercase", letterSpacing: "0.6px" }}>{label}</span>
        </div>
    );
}

// ─── Animated counter ─────────────────────────────────────────────────────────
function AnimatedNumber({ target, duration = 1200, suffix = "" }) {
    const [val, setVal] = useState(0);
    useEffect(() => {
        let start = null;
        const step = (ts) => {
            if (!start) start = ts;
            const p = Math.min((ts - start) / duration, 1);
            setVal(Math.round(p * target));
            if (p < 1) requestAnimationFrame(step);
        };
        requestAnimationFrame(step);
    }, [target, duration]);
    return <>{val}{suffix}</>;
}

// ─── Heatmap overlay renderer ─────────────────────────────────────────────────
function drawHeatmap(canvas, findings) {
    if (!canvas) return;
    const ctx = canvas.getContext("2d");
    const { width, height } = canvas;
    ctx.clearRect(0, 0, width, height);

    const regions = {
        consolidation: [{ x: 0.55, y: 0.45, r: 0.18, intensity: 0.75 }, { x: 0.65, y: 0.6, r: 0.13, intensity: 0.55 }],
        infiltrate: [{ x: 0.3, y: 0.4, r: 0.15, intensity: 0.5 }, { x: 0.45, y: 0.55, r: 0.1, intensity: 0.4 }],
        effusion: [{ x: 0.6, y: 0.75, r: 0.12, intensity: 0.6 }, { x: 0.35, y: 0.78, r: 0.1, intensity: 0.45 }],
        cardiomegaly: [{ x: 0.45, y: 0.52, r: 0.2, intensity: 0.45 }],
        normal: [],
    };

    findings.forEach((f) => {
        const pts = regions[f.id] || [];
        const c = f.color;
        pts.forEach(({ x, y, r, intensity }) => {
            const grd = ctx.createRadialGradient(x * width, y * height, 0, x * width, y * height, r * width);
            const rgb = c.replace("#", "");
            const rr = parseInt(rgb.slice(0, 2), 16);
            const gg = parseInt(rgb.slice(2, 4), 16);
            const bb = parseInt(rgb.slice(4, 6), 16);
            grd.addColorStop(0, `rgba(${rr},${gg},${bb},${intensity})`);
            grd.addColorStop(0.5, `rgba(${rr},${gg},${bb},${intensity * 0.4})`);
            grd.addColorStop(1, `rgba(${rr},${gg},${bb},0)`);
            ctx.fillStyle = grd;
            ctx.fillRect(0, 0, width, height);
        });
    });
}

                                    {/* Findings pills */}
                                    <div style={{ display: "flex", gap: 8, flexWrap: "wrap", marginTop: 16, paddingTop: 16, borderTop: "1px solid var(--border)" }}>
                                        {results.findings.map((f) => (

                                    </div>
                                    {hoveredFinding && (
                                        <p style={{ fontFamily: "'JetBrains Mono',monospace", fontSize: "0.68rem", color: "var(--text2)", marginTop: 10, transition: "all 0.2s" }}>

                                </div>

                                {/* Image viewer */}
                                <div className="viewer-card">

                                            </div>
                                        </>
                                    )}
                                </div>

                                <p className="disclaimer">
                                    ⚠ FOR RESEARCH USE ONLY · Not a certified medical device · Not a substitute for professional radiological interpretation<br />
                                    Always consult a qualified physician for clinical decisions
                                </p>
                            </div>
                        )}

                        {/* ── BOT TAB ── */}
                        {activeTab === "bot" && (
                            <div style={{ width: "100%", display: "flex", flexDirection: "column", gap: 16 }}>
                                {/* Chat window */}
                                <div style={{
                                    background: "var(--glass2)", border: "1px solid var(--border)", borderRadius: "var(--radius-lg)",
                                    backdropFilter: "blur(36px)", overflow: "hidden", display: "flex", flexDirection: "column",
                                    boxShadow: "0 24px 60px rgba(0,3,16,0.5)",

                                    {/* Chat header */}
                                    <div style={{ padding: "14px 20px", borderBottom: "1px solid var(--border)", display: "flex", alignItems: "center", gap: 12 }}>
                                        <div style={{
                                            width: 36, height: 36, borderRadius: 11, background: "linear-gradient(135deg,#2860ff,#0f2ea8)",

                                            <svg width="18" height="18" viewBox="0 0 24 24" fill="none" stroke="white" strokeWidth="2" strokeLinecap="round">

                                                {results ? `Context loaded: ${results.isPneumonia ? "Pneumonia detected" : "Normal study"}` : "General Q&A mode"}

                                        </div>
                                        <div className="badge-dot" style={{ marginLeft: "auto" }} />

                                                    color: msg.role === "user" ? "#fff" : "var(--text)",
                                                    fontFamily: "'Syne',sans-serif", fontSize: "0.84rem", lineHeight: 1.6,

                                            </div>
                                        ))}
                                        {chatLoading && (
                                            <div style={{ display: "flex" }}>
                                                <div style={{ padding: "12px 16px", borderRadius: "16px 16px 16px 4px", background: "rgba(90,150,255,0.07)", border: "1px solid var(--border)", display: "flex", gap: 5, alignItems: "center" }}>

                                                        <span key={i} style={{ width: 6, height: 6, borderRadius: "50%", background: "var(--accent)", display: "block", animation: `pulse-dot 1.2s ${i * 0.2}s ease-in-out infinite` }} />

                                                    fontFamily: "'JetBrains Mono',monospace", fontSize: "0.6rem",

                                        <input

                                            onChange={(e) => setChatInput(e.target.value)}
                                            onKeyDown={(e) => e.key === "Enter" && sendChat()}

                                            onClick={() => sendChat()}

                                            <svg width="16" height="16" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round"><line x1="22" y1="2" x2="11" y2="13" /><polygon points="22 2 15 22 11 13 2 9 22 2" /></svg>
                                        </button>
                                    </div>
                                </div>

                                {/* More quick questions */}
                                <div style={{ display: "flex", gap: 8, flexWrap: "wrap" }}>
                                    {QUICK_QUESTIONS.slice(3).map((q) => (
                                        <button
                                            key={q}
                                            onClick={() => sendChat(q)}
                                            style={{
                                                fontFamily: "'JetBrains Mono',monospace", fontSize: "0.62rem",
                                                padding: "7px 14px", borderRadius: 999, border: "1px solid var(--border2)",
                                                background: "rgba(90,150,255,0.04)", color: "var(--text2)", cursor: "pointer",
                                                transition: "all 0.2s",
                                            }}
                                            onMouseEnter={(e) => { e.currentTarget.style.background = "rgba(90,150,255,0.12)"; e.currentTarget.style.color = "var(--accent)"; }}
                                            onMouseLeave={(e) => { e.currentTarget.style.background = "rgba(90,150,255,0.04)"; e.currentTarget.style.color = "var(--text2)"; }}
                                        >
                                            {q}
                                        </button>
                                    ))}
                                </div>
                            </div>
                        )}

                        {/* ── INFO TAB ── */}
                        {activeTab === "info" && (
                            <div style={{ width: "100%", display: "flex", flexDirection: "column", gap: 16 }}>
                                {[
                                    {
                                        icon: "🫁", title: "What is Pneumonia?",
                                        body: "Pneumonia is an infection that inflames the air sacs (alveoli) in one or both lungs. The alveoli may fill with fluid or pus, causing cough with phlegm, fever, chills, and difficulty breathing.",
                                        color: "#5a96ff",
                                    },
                                    {
                                        icon: "🔬", title: "Types by Etiology",
                                        body: "Bacterial (S. pneumoniae most common), Viral (influenza, COVID-19, RSV), Fungal (Aspergillus, Histoplasma), Aspiration, Hospital-acquired (HAP), Ventilator-associated (VAP).",
                                        color: "#a855f7",
                                    },
                                    {
                                        icon: "📡", title: "Radiographic Findings",
                                        body: "Key signs: lobar or segmental consolidation, air bronchograms, ground-glass opacity (GGO), pleural effusion, silhouette sign, and Kerley B lines in pulmonary edema.",
                                        color: "#30d47a",
                                    },
                                    {
                                        icon: "⚠️", title: "When to Seek Emergency Care",
                                        body: "SpO₂ < 90%, respiratory rate > 30/min, confusion, persistent chest pain, systolic BP < 90 mmHg, or inability to maintain oral intake — these are CURB-65 criteria for severe pneumonia.",
                                        color: "#ff5b78",
                                    },
                                    {
                                        icon: "💊", title: "Treatment Overview",
                                        body: "Community-acquired: amoxicillin ± macrolide. Atypical: fluoroquinolone or doxycycline. Viral: supportive ± antivirals. Severe: IV beta-lactam + macrolide. Always guided by local antibiogram.",
                                        color: "#f59e0b",
                                    },
                                    {
                                        icon: "📈", title: "Prognosis & Recovery",
                                        body: "Outpatient mild: 5–7 days antibiotics, symptoms improve in 3–5 days. Radiographic clearance lags 4–6 weeks. Hospitalized patients: 7–14 day course. Elderly/immunocompromised may take longer.",
                                        color: "#06b6d4",
                                    },
                                ].map((card) => (
                                    <div key={card.title} style={{
                                        background: "var(--glass2)", border: "1px solid var(--border)", borderRadius: 18,
                                        padding: "18px 22px", backdropFilter: "blur(30px)",
                                        display: "flex", gap: 16, alignItems: "flex-start",
                                        transition: "all 0.25s", cursor: "default",
                                        position: "relative", overflow: "hidden",
                                    }}
                                        onMouseEnter={(e) => { e.currentTarget.style.borderColor = `${card.color}44`; e.currentTarget.style.background = `${card.color}06`; }}
                                        onMouseLeave={(e) => { e.currentTarget.style.borderColor = "var(--border)"; e.currentTarget.style.background = "var(--glass2)"; }}
                                    >
                                        <div style={{ width: 38, height: 38, borderRadius: 10, background: `${card.color}15`, border: `1px solid ${card.color}30`, display: "flex", alignItems: "center", justifyContent: "center", fontSize: "1.1rem", flexShrink: 0 }}>
                                            {card.icon}
                                        </div>
                                        <div>
                                            <p style={{ fontFamily: "'Syne',sans-serif", fontWeight: 700, fontSize: "0.9rem", color: card.color, marginBottom: 6 }}>{card.title}</p>
                                            <p style={{ fontFamily: "'Syne',sans-serif", fontSize: "0.82rem", color: "var(--text2)", lineHeight: 1.65 }}>{card.body}</p>
                                        </div>
                                    </div>
                                ))}
                            </div>
                        )}
                    </div>
                </section>
            )}

            {/* Footer */}
            <footer className="footer">
                <div className="footer-inner">
                    <span className="footer-text">Built with</span>

                    <span className="footer-amp">&</span>
                    <span className="footer-text">Powered by Claude · Research use only</span>
                </div>
            </footer>
        </div>
    );
